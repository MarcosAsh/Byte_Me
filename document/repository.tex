\section*{Overview}

This document describes all backend repositories, focusing on:
\begin{itemize}
    \item Custom query methods
    \item Entity relationships
\end{itemize}

\hrule
\vspace{1em}

\section{BadgeRepository}

\textbf{Entity:} Badge \\
\textbf{Extends:} \texttt{JpaRepository<Badge, UUID>}

\subsection*{Entity Relationships}
\begin{itemize}
    \item This repository manages the \texttt{Badge} entity.
    \item \texttt{Badge} is associated with \texttt{EmployeeBadge}, representing which users or employees have been awarded a badge.
\end{itemize}

\subsection*{Custom Query Methods}
\begin{itemize}
    \item \texttt{Optional<Badge> findByCode(String code)} \\
    Retrieves a badge using its unique business identifier (\texttt{code}).
\end{itemize}

\hrule
\vspace{1em}

\section{BundlePostingRepository}

\textbf{Entity:} BundlePosting \\
\textbf{Extends:} \texttt{JpaRepository<BundlePosting, UUID>}

\subsection*{Entity Relationships}
This repository manages the \texttt{BundlePosting} entity, which is associated with:
\begin{itemize}
    \item \texttt{Seller} (each bundle posting is created by a seller)
    \item \texttt{Category} (each bundle posting belongs to a category)
\end{itemize}

\subsection*{Custom Query Methods}
\begin{itemize}
    \item \texttt{List<BundlePosting> findBySeller\_SellerId(UUID sellerId)} \\
    Retrieves all bundle postings created by a seller.

    \item \texttt{Page<BundlePosting> findAvailable(Instant now, Pageable pageable)} \\
    Retrieves all active bundle postings that are not fully reserved and have not passed their pickup end time.

    \item \texttt{List<BundlePosting> findExpired(Instant now)} \\
    Retrieves all active bundle postings whose pickup end time has already passed.

    \item \texttt{long countBySeller(UUID sellerId)} \\
    Returns the total number of bundle postings created by a seller.
\end{itemize}

\section{EmployeeRepository}

\textbf{Entity:} Employee \\
\textbf{Extends:} \texttt{JpaRepository<Employee, UUID>}

\subsection*{Entity Relationships}
This repository manages the \texttt{Employee} entity, which is associated with:
\begin{itemize}
    \item \texttt{Organisation} (many employees belong to one organisation)
    \item \texttt{UserAccount} (an employee may be linked to a user account for login/access

    \item \texttt{EmployeeBadge} (tracks badges awarded to an employee)
    \item \texttt{Reservation} (reservations handled/managed by an employee)
    \item \texttt{RescueEvent} (events created or managed by an employee)
\end{itemize}

\subsection*{Custom Query Methods}
\begin{itemize}
    \item \texttt{Optional<Employee> findByUser\_UserId(UUID userId)} \\
    Retrieves the employee record linked to a \texttt{UserAccount}.

    \item \texttt{List<Employee> findByOrganisation\_OrgId(UUID orgId)} \\
    Retrieves all employees belonging to an organisation.
\end{itemize}

\hrule
\vspace{1em}

\section{EmployeeBadgeRepository}

\textbf{Entity:} EmployeeBadge \\
\textbf{Extends:} \texttt{JpaRepository<EmployeeBadge, EmployeeBadge.Key>}

\subsection*{Entity Relationships}
This repository manages the \texttt{EmployeeBadge} entity, representing a link between:
\begin{itemize}
    \item \texttt{Employee}
    \item \texttt{Badge} \\

    It is used to track which badges have been awarded to which employees.


\end{itemize}

\subsection*{Custom Query Methods}
\begin{itemize}
    \item \texttt{List<EmployeeBadge> findByEmployeeId(UUID employeeId)} \\
    Retrieves all badge assignments for a specific employee.

    \item \texttt{boolean existsByEmployeeIdAndBadgeId(UUID employeeId, UUID badgeId)} \\
    Checks whether a badge has already been awarded to an employee.
\end{itemize}

\section{IssueReportRepository}

\textbf{Entity:} IssueReport \\
\textbf{Extends:} \texttt{JpaRepository<IssueReport, UUID>}

\subsection*{Entity Relationships}
Associated with:
\begin{itemize}
    \item \texttt{BundlePosting} (the bundle posting that the issue refers to)
    \item \texttt{Seller} (indirectly via the associated bundle posting)
    \item \texttt{UserAccount} (the user who submitted the issue report)
    \item \texttt{Reservation} (the reservation related to the issue)
    \item \texttt{Employee} (the employee assigned to handle the issue)
\end{itemize}

\subsection*{Custom Query Methods}
\begin{itemize}
    \item \texttt{List<IssueReport> findOpenBySeller(UUID sellerId)} \\
    Retrieves all issue reports that are linked to bundle postings owned by a seller.



    
    \item \texttt{List<IssueReport> findBySeller(UUID sellerId)} \\
    Retrieves all issue reports associated with bundle postings owned by a seller.



    
\end{itemize}

\section{RescueEventRepository}

\textbf{Entity:} RescueEvent \\
\textbf{Extends:} \texttt{JpaRepository<RescueEvent, UUID>}

\subsection*{Entity Relationships}

This repository manages the \texttt{RescueEvent} entity, which is associated with:

\begin{itemize}
    \item \texttt{Employee} (many rescue events can be performed by a single employee)
    \item \texttt{Reservation} (each rescue event corresponds to a reservation)
\end{itemize}


\subsection*{Custom Query Methods}
\begin{itemize}
    \item \texttt{List<RescueEvent> findByEmployee\_EmployeeId(UUID employeeId)} \\
    Retrieves all rescue events performed by an employee.

    \item \texttt{long countByEmployee\_EmployeeId(UUID employeeId)} \\
    Returns the total number of rescue events performed by an employee.

    \item \texttt{long sumCo2eByEmployee(UUID employeeId)} \\
    Returns the total estimated CO$_2$e savings generated by rescue events performed by an employee.
    
    \item \texttt{long sumMealsByEmployee(UUID employeeId)} \\
    Returns the total estimated number of meals saved by rescue events performed by an employee.
\end{itemize}

\section{ReservationRepository}

\textbf{Entity:} Reservation \\
\textbf{Extends:} \texttt{JpaRepository<Reservation, UUID>}

\subsection*{Entity Relationships}

This repository manages the \texttt{Reservation} entity, which is associated with:

\begin{itemize}
    \item \texttt{Organisation} (the organisation that made the reservation)
    \item \texttt{UserAccount} (the user who created the reservation)
    \item \texttt{Employee} (the employee assigned to handle the reservation)
    \item \texttt{RescueEvent} (each reservation may result in a rescue event)
    \item \texttt{IssueReport} (a reservation can have multiple issue reports)
\end{itemize}


\subsection*{Custom Query Methods}
\begin{itemize}
    \item \texttt{List<Reservation> findByPosting\_PostingId(UUID postingId)} \\
    Retrieves all reservations for a bundle posting.


    \item \texttt{List<Reservation> findByOrganisation\_OrgId(UUID orgId)} \\
    Retrieves all reservations made by an organisation.


    \item \texttt{List<Reservation> findByEmployee\_EmployeeId(UUID employeeId)} \\
    Retrieves all reservations assigned to an employee.


    \item \texttt{Optional<Reservation> findByClaimCodeHash(String hash)} \\
    Retrieves a reservation using its hashed claim code


    \item \texttt{List<Reservation> findExpired(Instant now)} \\
    Retrieve bookings whose pickup deadline has passed.


    \item \texttt{long countByPosting\_PostingIdAndStatus(UUID postingId, \\ Reservation.Status status)} \\
    Returns the number of reservations for a given posting filtered by reservation status.


    \item \texttt{List<Reservation> findBySellerAndStatus(UUID sellerId, \\ Reservation.Status status)} \\
    Retrieves all reservations linked to bundle postings owned by a seller and filtered by status.


\end{itemize}

\section{UserAccountRepository}

\textbf{Entity:} UserAccount \\
\textbf{Extends:} \texttt{JpaRepository<UserAccount, UUID>}

\subsection*{Entity Relationships}

This repository manages the \texttt{UserAccount} entity, which is associated with:

\begin{itemize}
    \item \texttt{Seller} (a user account may be linked to a seller profile)
    \item \texttt {Employee} (a user account may be linked to an employee profile)
    \item \texttt{Reservation} (a user account can create multiple reservations)
    \item \texttt{IssueReport} (a user account can submit multiple issue reports)
\end{itemize}


\subsection*{Custom Query Methods}
\begin{itemize}
    \item \texttt{Optional<UserAccount> findByEmail(String email)} \\
    Retrieves a user account using its email address


    \item \texttt{boolean existsByEmail(String email)} \\
    Check if a user account for a given email address already exists.


\end{itemize}
